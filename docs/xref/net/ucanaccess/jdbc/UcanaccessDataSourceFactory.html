12  public class UcanaccessDataSourceFactory implements ObjectFactory {
13  
14      @Override
15      public Object getObjectInstance(Object uref, Name name, Context nameCtx, Hashtable<?, ?> environment) {
16  
17          String dataSourceClass = UcanaccessDataSource.class.getName();
18          Reference ref = (Reference) uref;
19          if (ref.getClassName().equals(dataSourceClass)) {
20              UcanaccessDataSource dataSource = new UcanaccessDataSource();
21              dataSource.setAccessPath((String) ref.get("accessPath").getContent());
22              dataSource.setUser((String) ref.get(Property.user.name()).getContent());
23              dataSource.setPassword((String) ref.get(Property.password.name()).getContent());
24  
25              dataSource.setColumnOrder((String) ref.get(Property.columnOrder.name()).getContent());
26              dataSource.setConcatNulls((Boolean) ref.get(Property.concatNulls.name()).getContent());
27              dataSource.setEncrypt((Boolean) ref.get(Property.encrypt.name()).getContent());
28              dataSource.setIgnoreCase((Boolean) ref.get(Property.ignoreCase.name()).getContent());
29              dataSource.setImmediatelyReleaseResources((Boolean) ref.get(Property.immediatelyReleaseResources.name()).getContent());
30              dataSource.setInactivityTimeout((Integer) ref.get(Property.inactivityTimeout.name()).getContent());
31              dataSource.setJackcessOpener((String) ref.get(Property.jackcessOpener.name()).getContent());
32              dataSource.setKeepMirror((String) ref.get(Property.keepMirror.name()).getContent());
33              dataSource.setLobScale((Integer) ref.get(Property.lobScale.name()).getContent());
34              dataSource.setMemory((Boolean) ref.get(Property.memory.name()).getContent());
35              dataSource.setMirrorFolder((String) ref.get(Property.mirrorFolder.name()).getContent());
36              dataSource.setNewDatabaseVersion((String) ref.get(Property.newDatabaseVersion.name()).getContent());
37              dataSource.setOpenExclusive((Boolean) ref.get(Property.openExclusive.name()).getContent());
38              dataSource.setPreventReloading((Boolean) ref.get(Property.preventReloading.name()).getContent());
39              dataSource.setReMap((String) ref.get(Property.reMap.name()).getContent());
40              dataSource.setShowSchema((Boolean) ref.get(Property.showSchema.name()).getContent());
41              dataSource.setSkipIndexes((Boolean) ref.get(Property.skipIndexes.name()).getContent());
42              dataSource.setSysSchema((Boolean) ref.get(Property.sysSchema.name()).getContent());
43  
44              return dataSource;
45          } else {
46              return null;
47          }
48      }
49  }
