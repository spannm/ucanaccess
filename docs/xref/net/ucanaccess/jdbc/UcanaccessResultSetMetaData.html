19      public UcanaccessResultSetMetaData(ResultSetMetaData _wrapped, Map<String, String> _aliases,
20              UcanaccessResultSet _resultSet) throws SQLException {
21          wrapped = _wrapped;
22

24          aliases = _aliases;

49          String name = SQLConverter.preEscapingIdentifier(wrapped.getColumnLabel(column));
50          return aliases.containsKey(name) ? aliases.get(name) : getColumnName(column);
51      }

53      @Override
54      public String getColumnName(int column) throws SQLException {
55          String columnName = SQLConverter.preEscapingIdentifier(wrapped.getColumnName(column));
56          String tableName = SQLConverter.preEscapingIdentifier(wrapped.getTableName(column));
57          String cn = metadata.getColumnName(tableName, columnName);
58          return cn == null ? columnName : cn;

60      }

89
