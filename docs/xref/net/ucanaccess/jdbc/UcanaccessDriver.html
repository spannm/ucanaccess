49      @Override
50      public Connection connect(String _url, Properties _props) throws SQLException {
51          if (!acceptsURL(_url)) {
52              return null;
53          }
54  
55          Map<String, String> unknownProps = new LinkedHashMap<>();
56          Map<Property, String> props = readProperties(_props, _url,
57              (k, v) -> {
58                  unknownProps.put(k, v);
59                  Logger.logWarning(LoggerMessageEnum.UNKNOWN_DRIVER_PROPERTY, k, v);
60              });
61  
62          int idxSemicolon = _url.indexOf(';');
63          String fileDbPath = idxSemicolon > 0 ? _url.substring(URL_PREFIX.length(), idxSemicolon) : _url.substring(URL_PREFIX.length());
64          File fileDb = new File(fileDbPath);
65          DBReferenceSingleton as = DBReferenceSingleton.getInstance();
66  
67          synchronized (UcanaccessDriver.class) {
68              try {
69                  Session session = new Session();
70  
71                  boolean alreadyLoaded = as.loaded(fileDb);
72                  FileFormat ff = null;
73                  if (props.containsKey(newDatabaseVersion)) {
74                      if (!fileDb.exists()) {
75                          ff = FileFormat.valueOf(props.get(newDatabaseVersion).toUpperCase());
76                      }
77  
78                  }
79                  boolean useCustomOpener = props.containsKey(jackcessOpener);
80  
81                  IJackcessOpenerInterface jko = useCustomOpener
82                      ? newJackcessOpenerInstance(props.get(jackcessOpener))
83                      : new DefaultJackcessOpener();
84                  DBReference dbRef = alreadyLoaded ? as.getReference(fileDb)
85                      : as.loadReference(fileDb, ff, jko, props.get(password));
86  
87                  if (!alreadyLoaded) {
88                      if ((useCustomOpener
89                          || props.containsKey(encrypt) && Boolean.parseBoolean(props.get(encrypt)))
90                          && (props.containsKey(memory) && !Boolean.parseBoolean(props.get(memory))
91                              || props.containsKey(keepMirror))) {
92                          dbRef.setEncryptHSQLDB(true);
93                      }
94  
95                      if (props.containsKey(memory)) {
96                          dbRef.setInMemory(Boolean.parseBoolean(props.get(memory)));
97                      }
98  
99                      if (props.containsKey(lobScale)) {
100                         Integer vl = validateLobScale(props.get(lobScale));
101                         dbRef.setLobScale(vl);
102                     }
103 
104                     if (props.containsKey(keepMirror)) {
105                         dbRef.setInMemory(false);
106                         if (dbRef.isEncryptHSQLDB()) {
107                             Logger.logWarning(LoggerMessageEnum.KEEP_MIRROR_AND_OTHERS);
108                         } else {
109                             File dbMirror =
110                                 new File(props.get(keepMirror) + fileDb.getName().toUpperCase().hashCode());
111                             dbRef.setToKeepHsql(dbMirror);
112                             if (props.containsKey(readOnlyMirror)) {
113                                 dbRef.setMirrorReadOnly(Boolean.parseBoolean(props.get(readOnlyMirror)));
114                             }
115                         }
116                     }
117 
118                     if (props.containsKey(showSchema)) {
119                         dbRef.setShowSchema(Boolean.parseBoolean(props.get(showSchema)));
120                     }
121                     if (props.containsKey(inactivityTimeout)) {
122                         int millis = 60000 * Integer.parseInt(props.get(inactivityTimeout));
123                         dbRef.setInactivityTimeout(millis);
124                     }
125                     if (props.containsKey(singleConnection)) {
126                         dbRef.setImmediatelyReleaseResources(Boolean.parseBoolean(props.get(singleConnection)));
127                     }
128                     if (props.containsKey(immediatelyReleaseResources)) {
129                         dbRef.setImmediatelyReleaseResources(
130                             Boolean.parseBoolean(props.get(immediatelyReleaseResources)));
131                     }
132                     if (props.containsKey(lockMdb)) {
133                         dbRef.setOpenExclusive(Boolean.parseBoolean(props.get(lockMdb)));
134                     }
135 
136                     if (props.containsKey(openExclusive)) {
137                         dbRef.setOpenExclusive(Boolean.parseBoolean(props.get(openExclusive)));
138                     }
139 
140                     if (props.containsKey(concatNulls)) {
141                         dbRef.setConcatNulls(Boolean.parseBoolean(props.get(concatNulls)));
142                     }
143                     if (props.containsKey(preventReloading)) {
144                         dbRef.setPreventReloading(Boolean.parseBoolean(props.get(preventReloading)));
145                     }
146                     if (props.containsKey(reMap)) {
147                         dbRef.setExternalResourcesMapping(toMap(props.get(reMap)));
148                     }
149                     if (props.containsKey(supportsAccessLike)) {
150                         SQLConverter.setSupportsAccessLike(Boolean.parseBoolean(props.get(supportsAccessLike)));
151                     }
152                     if (props.containsKey(columnOrder)
153                         && ColumnOrder.DISPLAY == ColumnOrder.parse(props.get(columnOrder))) {
154                         dbRef.setColumnOrderDisplay();
155                     }
156                     if (props.containsKey(mirrorFolder) && dbRef.getToKeepHsql() == null) {
157                         dbRef.setInMemory(false);
158                         String fd = props.get(mirrorFolder);
159                         if ("java.io.tmpdir".equals(fd)) {
160                             fd = System.getProperty("java.io.tmpdir");
161                         }
162                         dbRef.setMirrorFolder(new File(fd));
163                     }
164                     if (props.containsKey(ignoreCase)) {
165                         dbRef.setIgnoreCase(Boolean.parseBoolean(props.get(ignoreCase)));
166                     }
167 
168                     dbRef.getDbIO().setErrorHandler((cl, bt, location, ex) -> {
169                         if (cl.getType().isTextual()) {
170                             Logger.logWarning(LoggerMessageEnum.INVALID_CHARACTER_SEQUENCE,
171                                 cl.getTable().getName(), cl.getName(), new String(bt));
172                         }
173                         throw new IOException(ex.getMessage());
174                     });
175                 }
176                 String pwd = dbRef.getDbIO().getDatabasePassword();
177                 if (pwd != null && !props.containsKey(jackcessOpener)) {
178                     if (!pwd.equals(props.get(password))) {
179                         throw new UcanaccessSQLException(ExceptionMessages.NOT_A_VALID_PASSWORD);
180                     }
181 
182                 } else if (props.containsKey(jackcessOpener)) {
183                     String mpwd = props.get(password);
184                     session.setPassword(mpwd);
185                 }
186 
187                 Optional.ofNullable(props.get(user))
188                     .ifPresent(session::setUser);
189 
190                 SQLWarning sqlw = null;
191                 if (!alreadyLoaded) {
192                     boolean toBeLoaded = !dbRef.loadedFromKeptMirror(session);
193 
194                     Connection conn = dbRef.getHSQLDBConnection(session);
195                     // from version 2.7 hsqldb translates timestamps stored without timezone in the database
196                     // into the default timezone. MS Access however does not know timezones, therefore assume timestamps are UTC
197                     Try.withResources(conn::createStatement, st -> {
198                         st.executeQuery("SET TIME ZONE 'UTC'");
199                     }).orThrow();
200 
201                     LoadJet la = new LoadJet(conn, dbRef.getDbIO());
202                     Logger.turnOffJackcessLog();
203                     if (props.containsKey(sysSchema)) {
204                         boolean val = Boolean.parseBoolean(props.get(sysSchema));
205                         dbRef.setSysSchema(val);
206                         la.setSysSchema(val);
207 
208                     }
209                     if (props.containsKey(skipIndexes)) {
210                         boolean val = Boolean.parseBoolean(props.get(skipIndexes));
211                         dbRef.setSkipIndexes(val);
212                         la.setSkipIndexes(val);
213                     }
214 
215                     if (toBeLoaded) {
216                         la.loadDB();
217                     } else {
218                         la.resetFunctionsDefault();
219                     }
220                     as.put(fileDb.getAbsolutePath(), dbRef);
221                     sqlw = la.getLoadingWarnings();
222                 }
223 
224                 Properties newProps = new Properties();
225                 props.entrySet().stream()
226                     .forEach(e -> newProps.put(e.getKey().name(), e.getValue()));
227                 newProps.putAll(unknownProps);
228 
229                 UcanaccessConnection uc = new UcanaccessConnection(as.getReference(fileDb), newProps, session);
230                 uc.addWarnings(sqlw);
231                 uc.setUrl(_url);
232                 return uc;
233             } catch (Exception _ex) {
234                 throw new UcanaccessSQLException(_ex);
235             }
236         }
237     }

239     private Integer validateLobScale(String _property) {
240         try {
241             Integer i = Integer.parseInt(_property);
242 
243             if (i == 1 || i == 2 || i == 4 || i == 8 || i == 16 || i == 32) {
244                 return i;
245             }
246 
247         } catch (Exception _ignored) {
248 
249         }
250         Logger.logWarning(LoggerMessageEnum.LOBSCALE);
251         return null;
252     }

254     private Map<String, String> toMap(String property) {
255         Map<String, String> hm = new HashMap<>();
256         StringTokenizer st = new StringTokenizer(property, "&");
257         while (st.hasMoreTokens()) {
258             String entry = st.nextToken();
259             if (!entry.contains("|")) {
260                 continue;
261             }
262             hm.put(entry.substring(0, entry.indexOf('|')).toLowerCase(), entry.substring(entry.indexOf('|') + 1));
263         }
264         return hm;
265     }

292     private IJackcessOpenerInterface newJackcessOpenerInstance(String className)
293         throws InstantiationException, IllegalAccessException, ClassNotFoundException, UcanaccessSQLException, InvocationTargetException, NoSuchMethodException, SecurityException {
294         Object newInstance = Class.forName(className).getConstructor().newInstance();
295 
296         if (!IJackcessOpenerInterface.class.isInstance(newInstance)) {
297             throw new UcanaccessSQLException(ExceptionMessages.INVALID_JACKCESS_OPENER);
298         }
299         return (IJackcessOpenerInterface) newInstance;

302     /**
303      * Normalizes properties received as input and found in the driver url and returns a map of typed properties.<br>
304      * The method differentiates between known and unknown properties by return known properties as typed properties
305      * with their value and calling an optional consumer for each property key and value.<br>
306      * If a property is found multiple times or found in both, the input properties as well as in the driver url,
307      * the url has precedence over the input properties and the latest occurrence overall will be used.
308      *
309      * @param _input input properties
310      * @param _url driver url
311      * @param _unknownConsumer consumer of unknown property key/value
312      * @return map of unknown and normalized properties
313      */
314     static Map<Property, String> readProperties(Properties _input, String _url, BiConsumer<String, String> _unknownConsumer) {
315         Objects.requireNonNull(_input, "Properties required");
316         Objects.requireNonNull(_url, "URL required");
317 
318         Map<Property, String> props = new EnumMap<>(Property.class);
319 
320         for (String key : _input.stringPropertyNames()) {
321             Property prop = parse(key);
322             String val = _input.getProperty(key);
323             if (prop == null) {
324                 _unknownConsumer.accept(key, val);
325             } else {
326                 props.put(prop, val);
327             }
328         }
329 
330         Arrays.stream(_url.split(";"))
331             .skip(1)
332             .map(s -> s.split("=")).forEach(arr -> {
333                 Property prop = parse(arr[0]);
334                 String val = arr.length > 1 ? arr[1].strip() : null;
335                 if (prop == null) {
336                     _unknownConsumer.accept(arr[0], val);
337                 } else {
338                     props.put(prop, val);
339                 }
340             });
341 
342         return props;
343     }
