33      @Override
34      public Object executeWrapped() throws SQLException {
35          Statement w = getWrappedStatement();
36          switch (getCommandType()) {
37          case PREPARED_STATEMENT:
38              return ((PreparedStatement) w).execute();
39          case NO_ARGUMENTS:
40              return w.execute(getSql());
41          case WITH_COLUMNS_NAME:
42              return w.execute(getSql(), getColumnNames());
43          case WITH_AUTO_GENERATED_KEYS:
44              return w.execute(getSql(), getAutoGeneratedKeys());
45          case WITH_INDEXES:
46              return w.execute(getSql(), getIndexes());
47          default:
48              return false;
49          }
