14  public class DFunction {
15      private static final Pattern      FROM_PATTERN             = Pattern.compile("\\w*(?i)FROM\\w*");
16      private static final String       SELECT_FROM              = "(?i)SELECT(.*\\W)(?i)FROM(.*)";
17      private static final String       DFUNCTIONS_WHERE         =
18              "(?i)_[\\s\n\r]*\\([\\s\n\r]*['\"](.*)['\"]\\,[\\s\n\r]*['\"](.*)['\"]\\,[\\s\n\r]*['\"](.*)['\"][\\s\n\r]*\\)";
19      private static final String       DFUNCTIONS_WHERE_DYNAMIC =
20              "(?i)_[\\s\n\r]*\\([\\s\n\r]*['\"](.*)['\"]\\,[\\s\n\r]*['\"](.*)['\"]\\,(.*)\\)";
21      private static final String       DFUNCTIONS_NO_WHERE      =
22              "(?i)_[\\s\n\r]*\\([\\s\n\r]*['\"](.*)['\"]\\,[\\s\n\r]*['\"](.*)['\"][\\s\n\r]*\\)";
23      private static final String       IDENTIFIER               = "(\\W)((?i)_)(\\W)";
24      private static final List<String> DFUNCTIONLIST            =
25              List.of("COUNT", "MAX", "MIN", "SUM", "AVG", "LAST", "FIRST", "LOOKUP");
26  
27      private Connection                conn;
28      private String                    sql;

30      public DFunction(Connection _conn, String _sql) {
31          conn = _conn;
32          sql = _sql;

35      private String convertDFunctions() {
36          String sql0 = sql;
37          try {
38              boolean hasFrom = FROM_PATTERN.matcher(sql).find();
39              String init = hasFrom ? " (SELECT " : "";
40              String end = hasFrom ? " ) " : "";
41              for (String s : DFUNCTIONLIST) {
42  
43                  String fun = "D" + s;
44                  s = "lookup".equalsIgnoreCase(s) ? " " : s;
45                  sql0 = sql0.replaceAll(DFUNCTIONS_WHERE.replaceFirst("_", fun),
46                      init + s + "($1) FROM $2 WHERE $3     " + end);
47                  sql0 = sql0.replaceAll(DFUNCTIONS_NO_WHERE.replaceFirst("_", fun), init + s + "($1) FROM $2    " + end);
48                  Pattern dfd = Pattern.compile(DFUNCTIONS_WHERE_DYNAMIC.replaceFirst("_", fun));
49                  for (Matcher mtc = dfd.matcher(sql0); mtc.find(); mtc = dfd.matcher(sql0)) {
50                      StringBuilder sb = new StringBuilder();
51                      String g3 = mtc.group(3);
52                      String tableN = mtc.group(2).trim();
53                      String alias = tableN.startsWith("[") && tableN.endsWith("]") ? "[" + unpad(tableN) + "_DALIAS]"
54                          : tableN + "_DALIAS";
55                      String tn = tableN.startsWith("[") && tableN.endsWith("]") ? unpad(tableN) : tableN;
56                      sb.append(init).append(s).append("(").append(mtc.group(1)).append(") FROM ").append(tableN)
57                          .append(" AS ").append(alias).append(" WHERE ");
58                      boolean accessConcat = g3.indexOf('&') > 0;
59                      boolean sqlConcat = g3.indexOf("||") > 0;
60                      if (accessConcat || sqlConcat) {
61                          String concat = accessConcat ? "&" : Pattern.quote("||");
62                          String[] pts = g3.split(concat, -1);
63                          for (String tkn : pts) {
64                              if (isQuoted(tkn)) {
65                                  tkn = tkn.trim();
66                                  sb.append(unpad(tkn));
67                              } else {
68                                  tkn += " ";
69                                  for (String cln : getColumnNames(tn.toUpperCase())) {
70                                      String oppn = IDENTIFIER.replaceFirst("_", cln);
71                                      Pattern op = Pattern.compile(oppn);
72                                      Matcher mtcop = op.matcher(tkn);
73                                      if (!mtcop.find()) {
74                                          continue;
75                                      }
76                                      String pref = mtcop.group(1);
77                                      if (".".equals(pref) || "[".equals(pref) && mtcop.start(1) > 0
78                                          && tkn.charAt(mtcop.start(1) - 1) == '.') {
79                                          continue;
80                                      }
81                                      tkn = tkn.replaceAll(oppn,
82                                          "[".equals(pref) ? resolveAmbiguosTableName(cln) + ".$1$2$3"
83                                              : "$1" + resolveAmbiguosTableName(cln) + ".$2$3");
84                                  }
85                                  sb.append(tkn);
86                              }
87                          }
88                      }
89                      sb.append(end);
90                      sql0 = sql0.replaceFirst(DFUNCTIONS_WHERE_DYNAMIC.replaceFirst("_", fun), sb.toString());
91                  }
92              }
93          } catch (SQLException _ignored) {
94          }
95          return sql0;
96      }

98      private String resolveAmbiguosTableName(String _identifier) {
99          return Try.withResources(conn::createStatement, st -> {
100             String f4t = SQLConverter.convertSQL(
101                 sql.replaceAll("[\r\n]", " ").replaceFirst(SELECT_FROM, "SELECT " + _identifier + " FROM $2 ")).getSql();
102             ResultSetMetaData rsmd = st.executeQuery(f4t).getMetaData();
103             String tableN = rsmd.getTableName(1);
104             return tableN == null || tableN.isBlank() ? _identifier : tableN;
105         }).orElse(_identifier);
106     }

108     private List<String> getColumnNames(String tableName) throws SQLException {
109         List<String> ar = new ArrayList<>();
110         if (conn == null) {
111             UcanaccessConnection conu = UcanaccessConnection.getCtxConnection();
112             if (conu == null) {
113                 return ar;
114             }
115             conn = conu.getHSQLDBConnection();
116         }
117         DatabaseMetaData dbmd = conn.getMetaData();
118         ResultSet rs = dbmd.getColumns(null, null, tableName, null);
119         while (rs.next()) {
120             ar.add(rs.getString(COLUMN_NAME));
121         }
122         return ar;
123     }

125     private static boolean isQuoted(String g3) {
126         g3 = g3.trim();
127         return g3.startsWith("'") && g3.endsWith("'") && g3.substring(1, g3.length() - 1).indexOf('\'') < 0
128                 || g3.startsWith("\"") && g3.endsWith("\"") && g3.substring(1, g3.length() - 1).indexOf('"') < 0;
129     }
130 
131     private static String unpad(String tkn) {
132         return tkn.substring(1, tkn.length() - 1);
133     }

135     public String toSQL() {
136         return convertDFunctions();
137     }
