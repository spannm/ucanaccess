3   import static io.github.spannm.jackcess.impl.query.QueryFormat.NEWLINE;
4   
5   import io.github.spannm.jackcess.impl.query.AppendQueryImpl;
6   
7   import java.util.ArrayList;
8   import java.util.List;
9   
10  public class AppendQueryTemp extends AppendQueryImpl {
11  
12      public AppendQueryTemp(AppendQueryImpl impl) {
13          super(impl.getName(), impl.getRows(), impl.getObjectId(), impl.getObjectFlag());
14      }
15  
16      @Override
17      protected void toSQLString(StringBuilder builder) {
18          builder.append("INSERT INTO ");
19          toOptionalQuotedExpr(builder, getTargetTable(), true);
20          toRemoteDb(builder, getRemoteDbPath(), getRemoteDbType());
21          builder.append(NEWLINE);
22          List<String> values = getValues();
23          if (!values.isEmpty()) {
24              List<String> decl = getInsertDeclaration();
25              if (!decl.isEmpty()) {
26                  builder.append("(").append(getInsertDeclaration()).append(")");
27              }
28              builder.append(" VALUES (").append(values).append(')');
29          } else {
30              List<String> decl = getSelectDeclaration();
31              if (!decl.isEmpty()) {
32                  builder.append("(").append(getSelectDeclaration()).append(")");
33              }
34              toSQLSelectString(builder, true);
35          }
36      }
37  
38      private List<String> getInsertDeclaration() {
39          return new RowFormatter(getDeclaration(getValueRows())) {
40              @Override
41              protected void format(StringBuilder builder, Row row) {
42                  String column = row._name2;
43                  if (!(column.startsWith("[") && column.endsWith("]"))) {
44                      column = "[" + row._name2 + "]";
45                  }
46                  builder.append(column);
47              }
48          }.format();
49      }
50  
51      private List<String> getSelectDeclaration() {
52          return new RowFormatter(getDeclaration(getColumnRows())) {
53              @Override
54              protected void format(StringBuilder builder, Row row) {
55                  String column = row._name2;
56                  if (!(column.startsWith("[") && column.endsWith("]"))) {
57                      column = "[" + row._name2 + "]";
58                  }
59                  builder.append(column);
60              }
61          }.format();
62      }
63  
64      private List<Row> getDeclaration(List<Row> valueRows) {
65          List<Row> ardc = new ArrayList<>();
66          for (Row row : valueRows) {
67              if (row._name2 != null) {
68                  ardc.add(row);
69              }
70          }
71          return ardc;
72      }
