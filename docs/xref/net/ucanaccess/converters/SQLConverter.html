126     private static void aliases(String sql, NormalizedSQL nsql) {
127         Matcher m = Patterns.SELECT_FROM_START.matcher(sql);
128         if (m.find()) {
129 
130             int init = m.end();
131             sql = sql.substring(init);
132             m = Patterns.SELECT_FROM_END.matcher(sql);
133             if (m.find()) {
134                 int end = m.start();
135                 sql = sql.substring(0, end);
136 
137                 for (m = Patterns.UNESCAPED_ALIAS.matcher(sql); m.find(); m = Patterns.UNESCAPED_ALIAS.matcher(sql)) {
138                     int e = m.end();
139 
140                     sql = sql.substring(e) + " ";
141                     char[] sqlc = sql.toCharArray();
142                     if (sqlc[0] == '[') {
143                         continue;
144                     }
145                     StringBuilder sb = new StringBuilder();
146                     for (char c : sqlc) {
147                         if (c == ' ' || c == '\n' || c == '\r' || c == ',') {
148                             String key = preEscapingIdentifier(sb.toString());
149                             nsql.put(key, sb.toString());
150                             break;
151                         } else {
152                             sb.append(c);
153                         }
154                     }
155                 }
156 
157             }
158         }
159     }

175     private static int[] getQuoteGroup(String _s) {
176         if (!_s.contains("''")) {
177             Matcher m = Patterns.QUOTE_M.matcher(_s);
178             if (m.find()) {
179                 return new int[] {m.start(), m.end()};
180             }
181         } else {
182             int[] ret = new int[] {-1, -1};
183             Matcher m = Patterns.QUOTE_S.matcher(_s);
184             while (m.find()) {
185                 int start = m.start();
186                 int end = m.end();
187                 if ((end - start) % 2 == 0) {
188                     if (ret[0] == -1) {
189                         return new int[] {m.start(), m.end()};
190                     }
191                 } else {
192                     if (ret[0] == -1) {
193                         ret[0] = m.start();
194                     } else {
195                         ret[1] = m.end();
196                         return ret;
197                     }
198                 }
199             }
200         }
201         return new int[0];
202     }

204     private static int[] getDoubleQuoteGroup(String _s) {
205         if (!_s.contains("\"\"")) {
206             Matcher m = Patterns.DOUBLE_QUOTE_M.matcher(_s);
207             if (m.find()) {
208                 return new int[] {m.start(), m.end()};
209             }
210         } else {
211             int[] ret = new int[] {-1, -1};
212             Matcher mc = Patterns.DOUBLE_QUOTE_S.matcher(_s);
213             while (mc.find()) {
214                 int start = mc.start();
215                 int end = mc.end();
216                 if ((end - start) % 2 == 0) {
217                     if (ret[0] == -1) {
218                         return new int[] {mc.start(), mc.end()};
219                     }
220                     continue;
221                 } else {
222                     if (ret[0] == -1) {
223                         ret[0] = mc.start();
224                     } else {
225                         ret[1] = mc.end();
226                         return ret;
227                     }
228                 }
229             }
230         }
231         return new int[0];
232     }

372     public static NormalizedSQL convertSQL(String _sql, UcanaccessConnection _conn, boolean _creatingQuery) {
373         NormalizedSQL nsql = new NormalizedSQL();
374         String sql = _sql + " ";
375         aliases(sql, nsql);
376         sql = replaceBacktick(sql);
377         sql = replaceAposNames(sql);
378         sql = convertUnion(sql);
379         sql = convertAccessDate(sql);
380         sql = convertQuotedAliases(sql, nsql);
381         sql = escape(sql);
382         sql = convertLike(sql);
383         sql = replaceWhiteSpacedTables(sql);
384         sql = translateAccessPowerOperators(sql);
385         // sql = replaceExclamationPoints(sql);
386         if (!_creatingQuery) {
387             Pivot.checkAndRefreshPivot(sql, _conn);
388             sql = DFunction.convertDFunctions(sql, _conn);
389         }
390         sql = sql.trim();
391 
392         nsql.setSql(sql);
393 
394         return nsql;
395     }

424     private static String convertQuotedAliases(String sql, NormalizedSQL nsql) {
425         for (Matcher m = Patterns.KIND_OF_SUBQUERY.matcher(sql); m.find(); m = Patterns.KIND_OF_SUBQUERY.matcher(sql)) {
426             String g2 = m.group(2).trim();
427             if (g2.endsWith(";")) {
428                 g2 = g2.substring(0, g2.length() - 1);
429             }
430             sql = sql.substring(0, m.start()) + "(" + g2 + ")" + sql.substring(m.end());
431         }
432         Set<String> hs = new HashSet<>();
433         String sqle = sql;
434         String sqlN = "";
435         for (Matcher m = Patterns.QUOTED_ALIAS.matcher(sqle); m.find(); m = Patterns.QUOTED_ALIAS.matcher(sqle)) {
436             String g2 = m.group(2);
437             if (g2.indexOf('\'') >= 0 || g2.indexOf('"') >= 0) {
438                 hs.add(g2);
439             }
440             String value = g2.substring(1, g2.length() - 1);
441             nsql.put(preEscapingIdentifier(value), value);
442             sqlN += sqle.substring(0, m.start()) + m.group(1) + g2.replaceAll("['\"]", "") + m.group(3);
443             sqle = sqle.substring(m.end());
444         }
445         sql = sqlN + sqle;
446         for (String escaped : hs) {
447             sql = sql.replaceAll("\\[" + escaped.substring(1, escaped.length() - 1) + "\\]",
448                     escaped.replaceAll("['\"]", ""));
449         }
450         return sql;
451     }

457     static void addWhiteSpacedTableNames(String _name) {
458         if (_name == null) {
459             return;
460         }
461         String name = basicEscapingIdentifier(_name);
462         if (name == null || WHITE_SPACED_TABLE_NAMES.contains(name)) {
463             return;
464         }
465         for (String alrIn : WHITE_SPACED_TABLE_NAMES) {
466             if (name.contains(alrIn)) {
467                 WHITE_SPACED_TABLE_NAMES.add(WHITE_SPACED_TABLE_NAMES.indexOf(alrIn), name);
468                 return;
469             }
470         }
471         WHITE_SPACED_TABLE_NAMES.add(name);
472     }

482     public static String convertAccessDate(String sql) {

548     /**
549      * Translates MS Access SQL expressions using the '^' (power) operator
550      * into the standard SQL POWER(base, exponent) function.
551      * <p>
552      * This is a simplified implementation based on regular expressions and might not
553      * cover all edge cases or complex SQL constructs. For robust parsing and
554      * modification of SQL, we should use a dedicated SQL parser library e.g. JSqlParser.
555      *
556      * <p>The current implementation's {@code POWER_OPERAND_REGEX} is designed to identify:</p>
557      * <ul>
558      *   <li>Simple column names or identifiers (e.g., {@code myColumn}, {@code Table.Column}).</li>
559      *   <li>Numeric literals (integers, decimals, scientific notation, with optional sign).</li>
560      *   <li>Simple parenthesized expressions without nested parentheses (e.g., {@code (num + 1)}).</li>
561      *   <li>Parameter markers ({@code ?}) for use in {@code PreparedStatement} scenarios.</li>
562      * </ul>
563      * <p>It explicitly **does NOT reliably handle** operands that are:</p>
564      * <ul>
565      *   <li>Complex parenthesized expressions with nested parentheses (e.g., {@code (num + (val - 2))})</li>
566      *   <li>Function calls (e.g., {@code ABS(value)})</li>
567      *   <li>Subqueries</li>
568      *   <li>String literals containing the {@code ^} character</li>
569      *   <li>SQL comments containing the {@code ^} character</li>
570      * </ul>
571      * <p>It specifically targets patterns like {@code "operand ^ exponent"} where both operands
572      * strictly conform to the defined {@code POWER_OPERAND_REGEX}.</p>
573      *
574      * @param _sql the SQL string containing MS Access power operators.
575      * @return the translated SQL string using the POWER function.
576      */
577     static String translateAccessPowerOperators(String _sql) {
578         if (_sql == null || !_sql.contains("^")) {
579             return _sql;
580         }
581 
582         Matcher matcher = Patterns.POWER_OPERATION.matcher(_sql);
583         StringBuffer translatedSql = new StringBuffer();
584 
585         while (matcher.find()) {
586             String base = matcher.group(1);     // extract base operand
587             String exponent = matcher.group(2); // extract exponent operand
588 
589             // replace the matched "base ^ exponent" with "POWER(base, exponent)"
590             // appendReplacement is crucial for correctly handling multiple matches and non-matched parts
591             matcher.appendReplacement(translatedSql, "POWER(" + base + ", " + exponent + ")");
592         }
593         // append any remaining portion of the string after the last match
594         matcher.appendTail(translatedSql);
595 
596         return translatedSql.toString();
597     }
