5   import java.util.List;
6   
7   /**
8    * Enum of MS Access versions to encapsulate Jackcess {@link FileFormat}.
9    */
10  public enum AccessVersion {
11  
12      V2000(FileFormat.V2000),
13      V2003(FileFormat.V2003),
14      V2007(FileFormat.V2007),
15      V2010(FileFormat.V2010),
16      V2016(FileFormat.V2016);
17  
18      static final AccessVersion DEFAULT_ACCESS_VERSION = AccessVersion.V2003;
19  
20      private final FileFormat fileFormat;
21  
22      AccessVersion(FileFormat _fileFormat) {
23          fileFormat = _fileFormat;
24      }
25  
26      public FileFormat getFileFormat() {
27          return fileFormat;
28      }
29  
30      @Override
31      public String toString() {
32          return name().toLowerCase();
33      }
34  
35      public static List<AccessVersion> getDefaultAccessVersion() {
36          return List.of(DEFAULT_ACCESS_VERSION);
37      }
38  
39      public static AccessVersion parse(String _val) {
40          if (_val == null) {
41              return null;
42          }
43          String val = _val.strip().toLowerCase();
44          for (AccessVersion ver : values()) {
45              if (val.equals(ver.name())) {
46                  return ver;
47              }
48          }
49          return null;
50      }
51  
52  }
