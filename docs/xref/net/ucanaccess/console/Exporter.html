16  /**
17   * Exports a {@link ResultSet} to different formats, primarily CSV.
18   * <p>
19   * The class uses a builder pattern to configure the export options before calling its dump methods.
20   * <p>
21   * For example, to export a {@link ResultSet} to a CSV file with a custom delimiter:
22   * <pre>
23   * Exporter exporter = new Exporter.Builder()
24   *     .setDelimiter(",")
25   *     .build();
26   * ResultSet rs = ...;
27   * exporter.dumpCsv(rs, System.out);
28   * </pre>
29   * </p>
30   */
31  public final class Exporter {
32      /** The default delimiter is semi-colon for historical reasons. */
33      private static final String DEFAULT_CSV_DELIMITER = ";";
34  
35      private static final String QUOTE                 = "\"";
36  
37      // See http://unicode.org/faq/utf_bom.html#bom2
38      private static final byte[] UTF8_BYTE_ORDER_MARK  = {(byte) 0xEF, (byte) 0xBB, (byte) 0xBF};
39  
40      private final String        delimiter;
41      private final boolean       includeBom;
42      private final boolean       preserveNewlines;
43  
44      /** Builder for {@link Exporter}. */
45      public static class Builder {
46          private String  delimiter        = DEFAULT_CSV_DELIMITER;
47          private boolean includeBom       = false;
48          private boolean preserveNewlines = false;
49  
50          /** * Sets the CSV column delimiter. The default is a semi-colon (";").
51           *
52           * @param _delimiter The delimiter character to be used.
53           * @return The Builder instance for method chaining.
54           */
55          public Builder withDelimiter(String _delimiter) {
56              delimiter = _delimiter;
57              return this;
58          }
59  
60          /**
61           * Includes the Byte Order Mark (BOM) at the beginning of the output.
62           * <p>
63           * This is often needed by applications like Excel to correctly interpret UTF-8.
64           *
65           * @param _includeBom If {@code true}, the BOM will be included.
66           * @return The Builder instance for method chaining.
67           */
68          public Builder includeBom(boolean _includeBom) {
69              includeBom = _includeBom;
70              return this;
71          }
72  
73          /**
74           * Preserves embedded linefeed ({@code \n}) and carriage return ({@code \r}) characters in the output.
75           * <p>
76           * If this is set to {@code false}, newlines will be replaced by spaces.
77           *
78           * @param _preverseNewlines If {@code true}, newlines are preserved.
79           * @return The Builder instance for method chaining.
80           */
81          public Builder preserveNewlines(boolean _preverseNewlines) {
82              preserveNewlines = _preverseNewlines;
83              return this;
84          }
85  
86          public Exporter build() {
87              return new Exporter(delimiter, includeBom, preserveNewlines);
88          }
89      }
90  
91      private Exporter(String _delimter, boolean _includeBom, boolean _preserveNewlines) {
92          delimiter = _delimter;
93          includeBom = _includeBom;
94          preserveNewlines = _preserveNewlines;
95      }
96  
97      /**
98       * Prints the ResultSet {@code _rs} in CSV format to the output stream {@code _out}.
99       *
100      * @param _rs The ResultSet to be exported.
101      * @param _out The output stream to which the CSV data will be written.
102      * @throws SQLException if a database access error occurs.
103      * @throws IOException if an I/O error occurs while writing to the stream.
104      */
105     public void dumpCsv(ResultSet _rs, PrintStream _out) throws SQLException, IOException {
106 
107         // Print the UTF-8 byte order mark
108         if (includeBom) {
109             _out.write(UTF8_BYTE_ORDER_MARK);
110         }
111 
112         ResultSetMetaData meta = _rs.getMetaData();
113         int cols = meta.getColumnCount();
114 
115         // Print the CSV header row
116         String comma = "";
117         for (int i = 1; i <= cols; ++i) {
118             _out.print(comma);
119             _out.print(toCsv(meta.getColumnLabel(i), delimiter, preserveNewlines));
120             comma = delimiter;
121         }
122         _out.println();
123 
124         SimpleDateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
125         DecimalFormat decimalFormat = new DecimalFormat("0.0########");
126         DecimalFormatSymbols dfs = new DecimalFormatSymbols();
127         dfs.setDecimalSeparator('.');
128         decimalFormat.setDecimalFormatSymbols(dfs);
129         decimalFormat.setGroupingUsed(false);
130 
131         // print the resultset rows
132         while (_rs.next()) {
133             comma = "";
134             for (int i = 1; i <= cols; ++i) {
135                 Object o = _rs.getObject(i);
136                 String str;
137                 if (o == null) {
138                     str = "(null)";
139                 } else if (o.getClass().isArray()) {
140                     str = Arrays.toString((Object[]) o);
141                 } else if (o instanceof Date) {
142                     str = dateFormat.format((Date) o);
143                 } else if (o instanceof BigDecimal) {
144                     str = decimalFormat.format(o);
145                 } else {
146                     str = o.toString();
147                 }
148                 _out.print(comma);
149                 _out.print(toCsv(str, delimiter, preserveNewlines));
150                 comma = delimiter;
151             }
152             _out.println();
153         }
154     }

156     /**
157      * Prints the Google BigQuery schema of the table given by {@code _rs} in JSON format to the {@code _out} stream.<br/>
158      * See <a href="https://cloud.google.com/bigquery/bq-command-line-tool">bq command-line tool</a> for a description of the JSON schema format.
159      *
160      * @param _rs The ResultSet whose metadata will be used to generate the schema.
161      * @param _out The output stream to which the schema will be written.
162      * @throws SQLException if a database access error occurs.
163      */
164     public void dumpSchema(ResultSet _rs, PrintStream _out) throws SQLException {
165         ResultSetMetaData meta = _rs.getMetaData();
166         int cols = meta.getColumnCount();
167         _out.println('[');
168 
169         for (int i = 1; i <= cols; ++i) {
170             String name = meta.getColumnName(i);
171             int sqlType = meta.getColumnType(i);
172             int nullable = meta.isNullable(i);
173 
174             _out.print(toSchemaRow(name, sqlType, nullable));
175             _out.printf(i != cols ? ",%n" : "%n");
176         }
177         _out.println(']');
178     }

180     /**
181      * Returns the CSV representation of the given string {@code _str}.<br/>
182      * This method handles quoting and delimiter escaping according to a simplified CSV standard.
183      * <ul>
184      *   <li>A double-quote character (") is escaped by replacing it with two double-quotes ("").</li>
185      *   <li>If the string contains the delimiter, it is enclosed in double-quotes.</li>
186      *   <li>Newlines (`\r`, `\n`) are either preserved (if `_preserveNewlines` is `true`) or
187      *       replaced with a space character. Preserved newlines will also cause the string to be
188      *       enclosed in double-quotes.</li>
189      * </ul>
190      *
191      * @param _str The string to be formatted as CSV.
192      * @param _delimiter The delimiter character used for the CSV output.
193      * @param _preserveNewlines If `true`, newline characters are kept; otherwise, they are replaced by spaces.
194      * @return The CSV-formatted string.
195      */
196     static String toCsv(String _str, String _delimiter, boolean _preserveNewlines) {
197         boolean needsTextQualifier = false;
198 
199         // A double-quote is replaced with 2 double-quotes
200         if (_str.contains(QUOTE)) {
201             _str = _str.replace(QUOTE, QUOTE + QUOTE);
202             needsTextQualifier = true;
203         }
204 
205         // If the string contains the delimiter, wrap it in quotes
206         if (_str.contains(_delimiter)) {
207             needsTextQualifier = true;
208         }
209 
210         // Preserve or replace newlines
211         if (_preserveNewlines) {
212             needsTextQualifier = true;
213         } else {
214             _str = _str.replace('\n', ' ').replace('\r', ' ');
215         }
216 
217         if (needsTextQualifier) {
218             return QUOTE + _str + QUOTE;
219         } else {
220             return _str;
221         }
222     }

224     /** Returns one row of the BigQuery JSON schema file. */
225     static String toSchemaRow(String _name, int _sqlType, int _nullable) {
226         return String.format("{\"name\": \"%s\", \"type\": \"%s\", \"mode\": \"%s\"}", _name, toBigQueryType(_sqlType),
227             toBigQueryNullable(_nullable));
228     }

230     /**
231      * Maps the {@code java.sql.Types} values to BigQuery data types.<br/>
232      * We map to the BigQuery Standard SQL data types
233      * (<a href="https://cloud.google.com/bigquery/docs/reference/standard-sql/data-types">...</a>) instead of the legacy SQL data types
234      * (<a href="https://cloud.google.com/bigquery/data-types">...</a>).
235      *
236      * <p>
237      * Any JDBC type not explicitly defined will be mapped to a BigQuery "string" type.
238      *
239      * @param _sqlType The {@code java.sql.Types} integer value.
240      * @return The corresponding BigQuery data type as a string.
241      */
242     static String toBigQueryType(int _sqlType) {
243         switch (_sqlType) {
244         case Types.TINYINT:
245         case Types.SMALLINT:
246         case Types.INTEGER:
247         case Types.BIGINT:
248             return "int64";
249         case Types.FLOAT:
250         case Types.DOUBLE:
251         case Types.NUMERIC:
252         case Types.DECIMAL:
253             return "float64";
254         case Types.TIMESTAMP:
255             return "timestamp";
256         case Types.BOOLEAN:
257             return "bool";
258         default:
259             return "string";
260         }
261     }

263     /**
264      * Converts the {@code nullable} indicator from {@code ResultSetMetaData.isNullable()} to the equivalent BigQuery
265      * schema value.
266      *
267      * @param _nullable The {@code ResultSetMetaData.column*Nulls} value.
268      * @return The corresponding BigQuery schema value ("required" or "nullable").
269      */
270     static String toBigQueryNullable(int _nullable) {
271         switch (_nullable) {
272         case ResultSetMetaData.columnNoNulls:
273             return "required";
274         case ResultSetMetaData.columnNullable:
275         case ResultSetMetaData.columnNullableUnknown:
276             return "nullable";
277         default:
278             return "nullable";
279         }
280     }
