9   public class CreatePrimaryKeyCommand implements ICommand {
10      private String tableName;
11      private String execId;
12  
13      public CreatePrimaryKeyCommand(String _tableName, String _execId) {
14          tableName = _tableName;
15          execId = _execId;
16      }
17  
18      @Override
19      public String getExecId() {
20          return execId;
21      }
22  
23      @Override
24      public String getTableName() {
25          return tableName;
26      }
27  
28      @Override
29      public CommandType getType() {
30          return CommandType.DDL;
31      }
32  
33      @Override
34      public IFeedbackAction persist() throws SQLException {
35          Try.catching(() -> new Persist2Jet().createPrimaryKey(tableName))
36              .orThrow(UcanaccessSQLException::new);
37          return null;
38      }
39  
40      @Override
41      public IFeedbackAction rollback() {
42          return null;
43      }
44  }
