5   import java.util.Map;

7   public abstract class AbstractCursorCommand implements ICursorCommand {
8   
9       @Override
10      public boolean currentRowMatches(Cursor cur, Map<String, Object> currentRow) {
11          Map<String, Object> rowPattern = getRowPattern();
12          if (rowPattern.size() != currentRow.size()) {
13              return false;
14          }
15          for (Map.Entry<String, Object> e : currentRow.entrySet()) {
16              String columnName = e.getKey();
17              if (!cur.getColumnMatcher().matches(cur.getTable(), columnName, rowPattern.get(columnName), e.getValue())) {
18                  return false;
19              }
20          }
21          return true;
22      }
23  
24      public void replaceAutoincrement(Map<String, Object> map) {
25          getRowPattern().putAll(map);
26      }

28  }
