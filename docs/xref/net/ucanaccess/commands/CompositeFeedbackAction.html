4   import java.util.ArrayList;

7   public class CompositeFeedbackAction implements IFeedbackAction {
8       private final List<IFeedbackAction> actions = new ArrayList<>();
9   
10      @Override
11      public void doAction(ICommand toChange) throws SQLException {
12          for (IFeedbackAction action : actions) {
13              action.doAction(toChange);
14          }
15  
16      }
17  
18      public boolean add(IFeedbackAction ifa) {
19          if (ifa == null) {
20              return false;
21          }
22          return actions.add(ifa);
23      }

25  }
